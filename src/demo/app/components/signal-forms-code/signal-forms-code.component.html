Quick Reference

Imports

import { form, Field } from '@angular/forms/signals';
import { Validators } from '@angular/forms';

Creating a Form

const myForm = form({
  fieldName: [initialValue, validators]
});

Binding to AMW Components

<amw-input [field]="myForm.controls.fieldName"></amw-input>

Reading Values (Signals)

// All values
const values = myForm.value();

// Single field
const name = myForm.controls.name.value();

// Validity
const isValid = myForm.valid();
