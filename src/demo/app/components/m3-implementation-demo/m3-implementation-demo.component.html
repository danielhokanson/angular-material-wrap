Implementation Examples Code snippets for implementing M3 in your application

1. CSS Custom Properties (styles.scss)

/* Essential M3 Color Tokens */
:root {
  --mdc-theme-primary: #6750A4;
  --mdc-theme-on-primary: #1C1B1F;
  --mdc-theme-primary-container: #EADDFF;
  --mdc-theme-on-primary-container: #21005D;
  /* ... all other M3 tokens */
}

/* Dark Theme Support */
.dark-theme {
  --mdc-theme-primary: #D0BCFF;
  --mdc-theme-on-primary: #381E72;
  /* ... dark theme variants */
}

2. Typography Classes (HTML)

<h1 class="mdc-typography--headline-large">Main Heading</h1>
<h2 class="mdc-typography--headline-medium">Section Heading</h2>
<p class="mdc-typography--body-large">Body text content</p>
<span class="mdc-typography--label-medium">Form Label</span>

3. Component Usage (Angular)

<button mat-raised-button class="mat-mdc-raised-button">
  Primary Action
</button>

<mat-form-field class="mat-mdc-form-field">
  <input matInput placeholder="Input field">
</mat-form-field>

<mat-card class="mat-mdc-card">
  <mat-card-content>Card content</mat-card-content>
</mat-card>

4. Theme Switching (TypeScript)

// Toggle dark theme
toggleDarkTheme() {
  document.documentElement.classList.toggle('dark-theme');
}

// Apply custom theme
applyCustomTheme(primaryColor: string) {
  document.documentElement.style.setProperty('--mdc-theme-primary', primaryColor);
}
