Basic Usage

<!-- Full Calendar -->
<amw-calendar-full
  [events]="myEvents"
  [config]="calendarConfig"
  [currentDate]="selectedDate"
  (eventChange)="onEventChange($event)"
  (eventClick)="onEventClick($event)">
</amw-calendar-full>

<!-- Mini Calendar -->
<amw-calendar-mini
  [events]="myEvents"
  [config]="miniConfig"
  [showWeekView]="true"
  [compactMode]="false">
</amw-calendar-mini>

TypeScript Configuration

// Event interface
interface MyEvent {
  id: string;
  title: string;
  description?: string;
  category: string;
  priority: 'low' | 'medium' | 'high';
}

// Calendar configuration
const calendarConfig: CalendarConfig<MyEvent> = {
  displayProperty: 'title',
  editable: true,
  deletable: true,
  draggable: true,
  defaultDuration: 60,
  allowCreate: true,
  eventColor: (event) => event.data.priority === 'high' ? '#ba1a1a' : '#388e3c'
};

// Sample events
const myEvents: CalendarEvent<MyEvent>[] = [
  {
    id: '1',
    data: { id: '1', title: 'Meeting', category: 'work', priority: 'high' },
    start: new Date(2024, 0, 15, 10, 0),
    end: new Date(2024, 0, 15, 11, 0),
    title: 'Meeting',
    color: '#6750a4'
  }
];

Event Handlers

// Event change handler
onEventChange(event: CalendarEventChangeEvent<MyEvent>): void {
  switch (event.type) {
    case 'create':
      // Handle event creation
      break;
    case 'update':
      // Handle event update
      break;
    case 'delete':
      // Handle event deletion
      break;
    case 'move':
      // Handle event move
      break;
  }
}

// Event click handler
onEventClick(event: CalendarEvent<MyEvent>): void {
  console.log('Event clicked:', event.data.title);
}

// Event create handler
onEventCreate(event: { date: Date; time?: Date }): void {
  console.log('Create event at:', event.date);
}
